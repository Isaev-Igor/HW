//переписал с использованием тернарного оператора



function checkNumber(num) {
    let newArr = [];
    let rootNumber = Number.isInteger((Math.sqrt(num * 1)));

    newArr[0] = (!rootNumber && num > 1 && num % 2 !== 0 || num === 2) ? true : false;
    newArr[1] = (num % 2 == 0 && num !== 0) ? true : false;
    newArr[2] = (num % 10 === 0 && num !== 0) ? true : false;

    return console.log(newArr);
}


checkNumber(10)
