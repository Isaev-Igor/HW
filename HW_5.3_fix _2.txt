



function checkNumber(num) {
    for (let i = 2; i < num; i++) {
        if (num % i === 0) {
            if (num % 10 === 0 && num % 2 === 0) {
                return [false, true, true];
            } else if (num % 10 !== 0 && num % 2 === 0) {
                return [false, true, false]
            } else {
                return [false, false, false]
            }
        }
    }
    return [true, false, false];
}


console.log(checkNumber(11));
